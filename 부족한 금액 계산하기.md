Number.prototype.min = function(b){ return Math.min(this, b) }
Number.prototype.abs = function(){ return Math.abs(this) };
//monkey patch
function solution(price, money, count) { 
    return (money - (price * count * (1+ count) / 2))
        .min(0)
        .abs();
}
